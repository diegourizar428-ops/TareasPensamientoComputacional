n3= r1%10

if(num>99) and (num<1000):
    if (n1>n2): 
        aux = n1
        n1 = n2
        n2 = aux
    if(n2>n3):
        aux = n2
        n2 = n3
        n3 = aux
    if (n1>n2):
        aux = n1
        n1 = n2
        n2 = aux
else:
